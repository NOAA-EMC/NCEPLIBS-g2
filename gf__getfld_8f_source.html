1  subroutine gf_getfld(cgrib,lcgrib,ifldnum,unpack,expand,gfld,ierr)
2 !$$$ SUBPROGRAM DOCUMENTATION BLOCK
3 ! . . . .
4 ! SUBPROGRAM: gf_getfld
5 ! PRGMMR: Gilbert ORG: W/NP11 DATE: 2000-05-26
6 !
7 ! ABSTRACT: This subroutine returns the Grid Definition, Product Definition,
8 ! Bit-map ( if applicable ), and the unpacked data for a given data
9 ! field. All of the information returned is stored in a derived
10 ! type variable, gfld. Gfld is of type gribfield, which is defined
11 ! in module grib_mod, so users of this routine will need to include
12 ! the line "USE GRIB_MOD" in their calling routine. Each component of the
13 ! gribfield type is described in the OUTPUT ARGUMENT LIST section below.
14 !
15 ! Since there can be multiple data fields packed into a GRIB2
16 ! message, the calling routine indicates which field is being requested
17 ! with the ifldnum argument.
18 !
19 ! PROGRAM HISTORY LOG:
20 ! 2000-05-26 Gilbert
21 ! 2002-01-24 Gilbert - Changed to pass back derived type gribfield
22 ! variable through argument list, instead of
23 ! having many different arguments.
24 ! 2004-05-20 Gilbert - Added check to see if previous a bit-map is specified,
25 ! but none was found.
26 ! 2015-10-29 Vuong - Initial all pointers in derive type gribfield
27 !
28 ! USAGE: CALL gf_getfld(cgrib,lcgrib,ifldnum,unpack,expand,gfld,ierr)
29 ! INPUT ARGUMENT LIST:
30 ! cgrib - Character array that contains the GRIB2 message
31 ! lcgrib - Length (in bytes) of GRIB message array cgrib.
32 ! ifldnum - Specifies which field in the GRIB2 message to return.
33 ! unpack - Logical value indicating whether to unpack bitmap/data
34 ! .true. = unpack bitmap and data values
35 ! .false. = do not unpack bitmap and data values
36 ! expand - Boolean value indicating whether the data points should be
37 ! expanded to the correspond grid, if a bit-map is present.
38 ! 1 = if possible, expand data field to grid, inserting zero
39 ! values at gridpoints that are bitmapped out.
40 ! (SEE REMARKS2)
41 ! 0 = do not expand data field, leaving it an array of
42 ! consecutive data points for each "1" in the bitmap.
43 ! This argument is ignored if unpack == 0 OR if the
44 ! returned field does not contain a bit-map.
45 !
46 ! OUTPUT ARGUMENT LIST:
47 ! gfld - derived type gribfield ( defined in module grib_mod )
48 ! ( NOTE: See Remarks Section )
49 ! gfld%version = GRIB edition number ( currently 2 )
50 ! gfld%discipline = Message Discipline ( see Code Table 0.0 )
51 ! gfld%idsect() = Contains the entries in the Identification
52 ! Section ( Section 1 )
53 ! This element is actually a pointer to an array
54 ! that holds the data.
55 ! gfld%idsect(1) = Identification of originating Centre
56 ! ( see Common Code Table C-1 )
57 ! 7 - US National Weather Service
58 ! gfld%idsect(2) = Identification of originating Sub-centre
59 ! gfld%idsect(3) = GRIB Master Tables Version Number
60 ! ( see Code Table 1.0 )
61 ! 0 - Experimental
62 ! 1 - Initial operational version number
63 ! gfld%idsect(4) = GRIB Local Tables Version Number
64 ! ( see Code Table 1.1 )
65 ! 0 - Local tables not used
66 ! 1-254 - Number of local tables version used
67 ! gfld%idsect(5) = Significance of Reference Time (Code Table 1.2)
68 ! 0 - Analysis
69 ! 1 - Start of forecast
70 ! 2 - Verifying time of forecast
71 ! 3 - Observation time
72 ! gfld%idsect(6) = Year ( 4 digits )
73 ! gfld%idsect(7) = Month
74 ! gfld%idsect(8) = Day
75 ! gfld%idsect(9) = Hour
76 ! gfld%idsect(10) = Minute
77 ! gfld%idsect(11) = Second
78 ! gfld%idsect(12) = Production status of processed data
79 ! ( see Code Table 1.3 )
80 ! 0 - Operational products
81 ! 1 - Operational test products
82 ! 2 - Research products
83 ! 3 - Re-analysis products
84 ! gfld%idsect(13) = Type of processed data ( see Code Table 1.4 )
85 ! 0 - Analysis products
86 ! 1 - Forecast products
87 ! 2 - Analysis and forecast products
88 ! 3 - Control forecast products
89 ! 4 - Perturbed forecast products
90 ! 5 - Control and perturbed forecast products
91 ! 6 - Processed satellite observations
92 ! 7 - Processed radar observations
93 ! gfld%idsectlen = Number of elements in gfld%idsect().
94 ! gfld%local() = Pointer to character array containing contents
95 ! of Local Section 2, if included
96 ! gfld%locallen = length of array gfld%local()
97 ! gfld%ifldnum = field number within GRIB message
98 ! gfld%griddef = Source of grid definition (see Code Table 3.0)
99 ! 0 - Specified in Code table 3.1
100 ! 1 - Predetermined grid Defined by originating centre
101 ! gfld%ngrdpts = Number of grid points in the defined grid.
102 ! gfld%numoct_opt = Number of octets needed for each
103 ! additional grid points definition.
104 ! Used to define number of
105 ! points in each row ( or column ) for
106 ! non-regular grids.
107 ! = 0, if using regular grid.
108 ! gfld%interp_opt = Interpretation of list for optional points
109 ! definition. (Code Table 3.11)
110 ! gfld%igdtnum = Grid Definition Template Number (Code Table 3.1)
111 ! gfld%igdtmpl() = Contains the data values for the specified Grid
112 ! Definition Template ( NN=gfld%igdtnum ). Each
113 ! element of this integer array contains an entry (in
114 ! the order specified) of Grid Defintion Template 3.NN
115 ! This element is actually a pointer to an array
116 ! that holds the data.
117 ! gfld%igdtlen = Number of elements in gfld%igdtmpl(). i.e. number of
118 ! entries in Grid Defintion Template 3.NN
119 ! ( NN=gfld%igdtnum ).
120 ! gfld%list_opt() = (Used if gfld%numoct_opt .ne. 0) This array
121 ! contains the number of grid points contained in
122 ! each row ( or column ). (part of Section 3)
123 ! This element is actually a pointer to an array
124 ! that holds the data. This pointer is nullified
125 ! if gfld%numoct_opt=0.
126 ! gfld%num_opt = (Used if gfld%numoct_opt .ne. 0) The number of entries
127 ! in array ideflist. i.e. number of rows ( or columns )
128 ! for which optional grid points are defined. This value
129 ! is set to zero, if gfld%numoct_opt=0.
130 ! gfdl%ipdtnum = Product Definition Template Number (see Code Table 4.0)
131 ! gfld%ipdtmpl() = Contains the data values for the specified Product
132 ! Definition Template ( N=gfdl%ipdtnum ). Each element
133 ! of this integer array contains an entry (in the
134 ! order specified) of Product Defintion Template 4.N.
135 ! This element is actually a pointer to an array
136 ! that holds the data.
137 ! gfld%ipdtlen = Number of elements in gfld%ipdtmpl(). i.e. number of
138 ! entries in Product Defintion Template 4.N
139 ! ( N=gfdl%ipdtnum ).
140 ! gfld%coord_list() = Real array containing floating point values
141 ! intended to document the vertical discretisation
142 ! associated to model data on hybrid coordinate
143 ! vertical levels. (part of Section 4)
144 ! This element is actually a pointer to an array
145 ! that holds the data.
146 ! gfld%num_coord = number of values in array gfld%coord_list().
147 ! gfld%ndpts = Number of data points unpacked and returned.
148 ! gfld%idrtnum = Data Representation Template Number
149 ! ( see Code Table 5.0)
150 ! gfld%idrtmpl() = Contains the data values for the specified Data
151 ! Representation Template ( N=gfld%idrtnum ). Each
152 ! element of this integer array contains an entry
153 ! (in the order specified) of Product Defintion
154 ! Template 5.N.
155 ! This element is actually a pointer to an array
156 ! that holds the data.
157 ! gfld%idrtlen = Number of elements in gfld%idrtmpl(). i.e. number
158 ! of entries in Data Representation Template 5.N
159 ! ( N=gfld%idrtnum ).
160 ! gfld%unpacked = logical value indicating whether the bitmap and
161 ! data values were unpacked. If false,
162 ! gfld%bmap and gfld%fld pointers are nullified.
163 ! gfld%expanded = Logical value indicating whether the data field
164 ! was expanded to the grid in the case where a
165 ! bit-map is present. If true, the data points in
166 ! gfld%fld match the grid points and zeros were
167 ! inserted at grid points where data was bit-mapped
168 ! out. If false, the data values in gfld%fld were
169 ! not expanded to the grid and are just a consecutive
170 ! array of data points corresponding to each value of
171 ! "1" in gfld%bmap.
172 ! gfld%ibmap = Bitmap indicator ( see Code Table 6.0 )
173 ! 0 = bitmap applies and is included in Section 6.
174 ! 1-253 = Predefined bitmap applies
175 ! 254 = Previously defined bitmap applies to this field
176 ! 255 = Bit map does not apply to this product.
177 ! gfld%bmap() = Logical*1 array containing decoded bitmap,
178 ! if ibmap=0 or ibap=254. Otherwise nullified.
179 ! This element is actually a pointer to an array
180 ! that holds the data.
181 ! gfld%fld() = Array of gfld%ndpts unpacked data points.
182 ! This element is actually a pointer to an array
183 ! that holds the data.
184 ! ierr - Error return code.
185 ! 0 = no error
186 ! 1 = Beginning characters "GRIB" not found.
187 ! 2 = GRIB message is not Edition 2.
188 ! 3 = The data field request number was not positive.
189 ! 4 = End string "7777" found, but not where expected.
190 ! 6 = GRIB message did not contain the requested number of
191 ! data fields.
192 ! 7 = End string "7777" not found at end of message.
193 ! 8 = Unrecognized Section encountered.
194 ! 9 = Data Representation Template 5.NN not yet implemented.
195 ! 15 = Error unpacking Section 1.
196 ! 16 = Error unpacking Section 2.
197 ! 10 = Error unpacking Section 3.
198 ! 11 = Error unpacking Section 4.
199 ! 12 = Error unpacking Section 5.
200 ! 13 = Error unpacking Section 6.
201 ! 14 = Error unpacking Section 7.
202 ! 17 = Previous bitmap specified, but none exists.
203 !
204 ! REMARKS: Note that derived type gribfield contains pointers to many
205 ! arrays of data. The memory for these arrays is allocated
206 ! when the values in the arrays are set, to help minimize
207 ! problems with array overloading. Because of this users
208 ! are encouraged to free up this memory, when it is no longer
209 ! needed, by an explicit call to subroutine gf_free.
210 ! ( i.e. CALL GF_FREE(GFLD) )
211 !
212 ! Subroutine gb_info can be used to first determine
213 ! how many data fields exist in a given GRIB message.
214 !
215 ! REMARKS2: It may not always be possible to expand a bit-mapped data field.
216 ! If a pre-defined bit-map is used and not included in the GRIB2
217 ! message itself, this routine would not have the necessary
218 ! information to expand the data. In this case, gfld%expanded would
219 ! would be set to 0 (false), regardless of the value of input
220 ! argument expand.
221 !
222 ! ATTRIBUTES:
223 ! LANGUAGE: Fortran 90
224 ! MACHINE: IBM SP
225 !
226 !$$$
227  use grib_mod

304  character(len=1),intent(in) :: cgrib(lcgrib)
305  integer,intent(in) :: lcgrib,ndpts,idrsnum,igdsnum
306  integer,intent(inout) :: iofst

313  have3=.false.
314  have4=.false.
315  have5=.false.
316  have6=.false.
317  have7=.false.
318  ierr=0
319  numfld=0
320  gfld%locallen=0
321  nullify(gfld%list_opt,gfld%igdtmpl,gfld%ipdtmpl)
322  nullify(gfld%coord_list,gfld%idrtmpl,gfld%bmap,gfld%fld)
323 !
324 ! Check for valid request number
325 !
326  if (ifldnum.le.0) then
327  print *,'gf_getfld: Request for field number must be positive.'
328  ierr=3
329  return
330  endif
331 !
332 ! Check for beginning of GRIB message in the first 100 bytes
333 !
334  istart=0
335  do j=1,100
336  ctemp=cgrib(j)//cgrib(j+1)//cgrib(j+2)//cgrib(j+3)
337  if (ctemp.eq.grib ) then
338  istart=j
339  exit
340  endif
341  enddo
342  if (istart.eq.0) then
343  print *,'gf_getfld: Beginning characters GRIB not found.'
344  ierr=1
345  return
346  endif
347 !
348 ! Unpack Section 0 - Indicator Section
349 !
350  iofst=8*(istart+5)
351  call g2_gbytec(cgrib,listsec0(1),iofst,8) ! Discipline
352  iofst=iofst+8
353  call g2_gbytec(cgrib,listsec0(2),iofst,8) ! GRIB edition number
354  iofst=iofst+8
355  iofst=iofst+32
356  call g2_gbytec(cgrib,lengrib,iofst,32) ! Length of GRIB message
357  iofst=iofst+32
358  lensec0=16
359  ipos=istart+lensec0
360 !
361 ! Currently handles only GRIB Edition 2.
362 !
363  if (listsec0(2).ne.2) then
364  print *,'gf_getfld: can only decode GRIB edition 2.'
365  ierr=2
366  return
367  endif
368 !
369 ! Loop through the remaining sections keeping track of the
370 ! length of each. Also keep the latest Grid Definition Section info.
371 ! Unpack the requested field number.
372 !
373  do
374  ! Check to see if we are at end of GRIB message
375  ctemp=cgrib(ipos)//cgrib(ipos+1)//cgrib(ipos+2)//cgrib(ipos+3)
376  if (ctemp.eq.c7777 ) then
377  ipos=ipos+4
378  ! If end of GRIB message not where expected, issue error
379  if (ipos.ne.(istart+lengrib)) then
380  print *,'gf_getfld: "7777" found, but not where expected.'
381  ierr=4
382  return
383  endif
384  exit
385  endif
386  ! Get length of Section and Section number
387  iofst=(ipos-1)*8
388  call g2_gbytec(cgrib,lensec,iofst,32) ! Get Length of Section
389  iofst=iofst+32
390  call g2_gbytec(cgrib,isecnum,iofst,8) ! Get Section number
391  iofst=iofst+8
392  !print *,' lensec= ',lensec,' secnum= ',isecnum
393  !
394  ! Check to see if section number is valid
395  !
396  if ( (isecnum.lt.1).OR.(isecnum.gt.7) ) then
397  print *,'gf_getfld: Unrecognized Section Encountered=',isecnum
398  ierr=8
399  return
400  endif
401  !
402  ! If found Section 1, decode elements in Identification Section
403  !
404  if (isecnum.eq.1) then
405  iofst=iofst-40 ! reset offset to beginning of section
406  call gf_unpack1(cgrib,lcgrib,iofst,gfld%idsect,
407  & gfld%idsectlen,jerr)
408  if (jerr.ne.0) then
409  ierr=15
410  return
411  endif
412  endif
413  !
414  ! If found Section 2, Grab local section
415  ! Save in case this is the latest one before the requested field.
416  !
417  if (isecnum.eq.2) then
418  iofst=iofst-40 ! reset offset to beginning of section
419  if (associated(gfld%local)) deallocate(gfld%local)
420  call gf_unpack2(cgrib,lcgrib,iofst,gfld%locallen,
421  & gfld%local,jerr)
422  if (jerr.ne.0) then
423  ierr=16
424  return
425  endif
426  endif
427  !
428  ! If found Section 3, unpack the GDS info using the
429  ! appropriate template. Save in case this is the latest
430  ! grid before the requested field.
431  !
432  if (isecnum.eq.3) then
433  iofst=iofst-40 ! reset offset to beginning of section
434  if (associated(gfld%igdtmpl)) deallocate(gfld%igdtmpl)
435  if (associated(gfld%list_opt)) deallocate(gfld%list_opt)
436  call gf_unpack3(cgrib,lcgrib,iofst,igds,gfld%igdtmpl,
437  & gfld%igdtlen,gfld%list_opt,gfld%num_opt,jerr)
438  if (jerr.eq.0) then
439  have3=.true.
440  gfld%griddef=igds(1)
441  gfld%ngrdpts=igds(2)
442  gfld%numoct_opt=igds(3)
443  gfld%interp_opt=igds(4)
444  gfld%igdtnum=igds(5)
445  else
446  ierr=10
447  return
448  endif
449  endif
450  !
451  ! If found Section 4, check to see if this field is the
452  ! one requested.
453  !
454  if (isecnum.eq.4) then
455  numfld=numfld+1
456  if (numfld.eq.ifldnum) then
457  gfld%discipline=listsec0(1)
458  gfld%version=listsec0(2)
459  gfld%ifldnum=ifldnum
460  gfld%unpacked=unpack
461  gfld%expanded=.false.
462  iofst=iofst-40 ! reset offset to beginning of section
463  call gf_unpack4(cgrib,lcgrib,iofst,gfld%ipdtnum,
464  & gfld%ipdtmpl,gfld%ipdtlen,gfld%coord_list,
465  & gfld%num_coord,jerr)
466  if (jerr.eq.0) then
467  have4=.true.
468  else
469  ierr=11
470  return
471  endif
472  endif
473  endif
474  !
475  ! If found Section 5, check to see if this field is the
476  ! one requested.
477  !
478  if ((isecnum.eq.5).and.(numfld.eq.ifldnum)) then
479  iofst=iofst-40 ! reset offset to beginning of section
480  call gf_unpack5(cgrib,lcgrib,iofst,gfld%ndpts,gfld%idrtnum,
481  & gfld%idrtmpl,gfld%idrtlen,jerr)
482  if (jerr.eq.0) then
483  have5=.true.
484  else
485  ierr=12
486  return
487  endif
488  endif
489  !
490  ! If found Section 6, Unpack bitmap.
491  ! Save in case this is the latest
492  ! bitmap before the requested field.
493  !
494  if (isecnum.eq.6) then
495  if (unpack) then ! unpack bitmap
496  iofst=iofst-40 ! reset offset to beginning of section
497  bmpsave=>gfld%bmap ! save pointer to previous bitmap
498  call gf_unpack6(cgrib,lcgrib,iofst,gfld%ngrdpts,gfld%ibmap,
499  & gfld%bmap,jerr)
500  if (jerr.eq.0) then
501  have6=.true.
502  if (gfld%ibmap .eq. 254) then ! use previously specified bitmap
503  if ( associated(bmpsave) ) then
504  gfld%bmap=>bmpsave
505  else
506  print *,'gf_getfld: Previous bit-map specified,',
507  & ' but none exists,'
508  ierr=17
509  return
510  endif
511  else ! get rid of it
512  if ( associated(bmpsave) ) deallocate(bmpsave)
513  endif
514  else
515  ierr=13
516  return
517  endif
518  else ! do not unpack bitmap
519  call g2_gbytec(cgrib,gfld%ibmap,iofst,8) ! Get BitMap Indicator
520  have6=.true.
521  endif
522  endif
523  !
524  ! If found Section 7, check to see if this field is the
525  ! one requested.
526  !
527  if ((isecnum.eq.7).and.(numfld.eq.ifldnum).and.unpack) then
528  iofst=iofst-40 ! reset offset to beginning of section
529  call gf_unpack7(cgrib,lcgrib,iofst,gfld%igdtnum,
530  & gfld%igdtmpl,gfld%idrtnum,
531  & gfld%idrtmpl,gfld%ndpts,
532  & gfld%fld,jerr)
533  if (jerr.eq.0) then
534  have7=.true.
535  ! If bitmap is used with this field, expand data field
536  ! to grid, if possible.
537  if ( gfld%ibmap .ne. 255 .AND. associated(gfld%bmap) ) then
538  if ( expand ) then
539  allocate(newfld(gfld%ngrdpts))
540  !newfld(1:gfld%ngrdpts)=0.0
